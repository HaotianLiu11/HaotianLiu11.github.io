struct Student{
char num;//学号
char name[25];//姓名
char major[10];//专业
int classNo;//班级序号（1、2）
int score[3];//课程成绩
};
int N;
typedef struct Student STU;
void Input(STU p);//从键盘输入N个学生的信息
void Output(STU p);//将某个学生信息格式化输出
void Save(STU p);//将某个学生信息存入文件
int Fetch(STU st,int n,char filename);//从文件中随机读取某个学生信息
void Max(STU stu[],int m);//求所有学生某门课程的最高分和分数最高的学生的姓名。
double average_person(STU p);
void sort_select(STU p);//对某个专业的学生，按总平均成绩由低到高进行简单选择排序。
void Sort_Buble(STU p);//对某个班级的学生，按总平均成绩由高到低进行起泡排序。
void Sort_insert(STU p);//对某个班级的学生，按某门课程成绩由低到高进行直接插入排序。
void Search(STU p);//实现班级和成绩的综合查找
void Ask();

}

//——————————————————————————随机读取—————————————————————————————-
int Fetch(STU p,int n,char filename)
{
FILE fp;
if((fp=fopen(filename,”rb”))==NULL)
{
fclose(fp);
return 0;
}
fseek(fp,(long)sizeof(STU)n,0);//将位置指针从文件的开始往后移n个STU结构体的长度,n是提取的位置。
if ( (fread(p,sizeof(STU),1,fp)) != 1)//从文件中一次读取一个结构体量值
{
printf(“文件录入失败 请重新输入：”);
fclose(fp);
return 0;
}
Output(p);
fclose(fp);
return 1;
}

void Sort_insert(STU p)
{
int i,j,id,q,tem;
int score[999];
char b[10];
int k=0; //用于计算学生数
STU m;
m=p;

gets(b);
printf (“请输入需要查看成绩的班级（请输入已经录入的班级）：”);
scanf (“%d”,&q);
printf (“请输入需要查看分数的课程（1,2,3）\n”);
scanf (“%d”,&id);
getchar();
for (i=0; imajor,b)==0)
{ if(p->classNo==q){
score[k]=p->score[id-1];
k++; //计算所选出的学生数
}
}
}

for (i=1; i<k; i++)
    if (score[i-1]>score[i])
    {
        tem=score[i];//等于小的那个数字
        j=i-1;//记录大的数的位置
        do
        {
            score[j+1]=score[j];//把大的数赋值给小的数,相当于把大的数字向后推了一位，但是原位置的数字不动
            j--;
        }
        while (j>=0&&tem<score[j]);  //把tem和较大的数的前一位比较，如果比它小，就让大的数在往后推一个
        score[j+1]=tem;              //把tem插入到j和j+1之间的位置。
    }

printf ("对该班级的学生，按第%d科成绩由低到高进行直接插入排序为:\n",id);
for (i=0; i<k; i++)
{
     p=m;
    for (j=0; j<N; j++,p++)
    {

        if (strcmp(p->major,b)==0 && (p->score[id]==score[i]) && (p->classNo==q))//并且该学生成绩的这门成绩等于成绩数组的第i个数字
        {
            Output(p);
            printf("\n");
        }
    }
}

}
